194 static inline double
195 li_fixed_to_double (li_fixed_t f)
196 {
197  union {
198  double d;
199  int64_t i;
200  } u;
201 
202  u.i = ((1023LL + 44LL) << 52) + (1LL << 51) + f;
203 
204  return u.d - (3LL << 43);
205 }
